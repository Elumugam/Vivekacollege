Run these SQL statements in your Supabase project's SQL editor to create the CMS tables used by the server.

-- Create generic content tables
CREATE TABLE IF NOT EXISTS homepage_content (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  key text UNIQUE,
  content jsonb,
  updated_by bigint,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

CREATE TABLE IF NOT EXISTS about_content (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  key text UNIQUE,
  content jsonb,
  updated_by bigint,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

CREATE TABLE IF NOT EXISTS contact_content (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  key text UNIQUE,
  content jsonb,
  updated_by bigint,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

CREATE TABLE IF NOT EXISTS popup_banners (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  key text UNIQUE,
  content jsonb,
  enabled boolean DEFAULT false,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

-- Courses table
CREATE TABLE IF NOT EXISTS courses (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  title text,
  slug text UNIQUE,
  category text,
  image text,
  video text,
  duration text,
  eligibility text,
  fees text,
  seats integer DEFAULT 0,
  description text,
  highlights text,
  is_featured boolean DEFAULT false,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

-- Gallery table
CREATE TABLE IF NOT EXISTS gallery (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  title text,
  description text,
  category text,
  type text,
  src text,
  is_published boolean DEFAULT true,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

-- Media assets table
CREATE TABLE IF NOT EXISTS media_assets (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  name text,
  url text,
  bucket text,
  meta jsonb,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

-- Admins table (for server auth)
CREATE TABLE IF NOT EXISTS admins (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  name text,
  email text UNIQUE,
  role text DEFAULT 'admin',
  password text,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

-- Trigger to update updated_at
CREATE OR REPLACE FUNCTION trigger_set_timestamp()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DO $$
BEGIN
  IF NOT EXISTS (SELECT 1 FROM pg_trigger WHERE tgname = 'set_timestamp') THEN
    -- attach trigger to desired tables
    EXECUTE 'CREATE TRIGGER set_timestamp BEFORE UPDATE ON homepage_content FOR EACH ROW EXECUTE FUNCTION trigger_set_timestamp()';
    EXECUTE 'CREATE TRIGGER set_timestamp BEFORE UPDATE ON about_content FOR EACH ROW EXECUTE FUNCTION trigger_set_timestamp()';
    EXECUTE 'CREATE TRIGGER set_timestamp BEFORE UPDATE ON contact_content FOR EACH ROW EXECUTE FUNCTION trigger_set_timestamp()';
    EXECUTE 'CREATE TRIGGER set_timestamp BEFORE UPDATE ON popup_banners FOR EACH ROW EXECUTE FUNCTION trigger_set_timestamp()';
    EXECUTE 'CREATE TRIGGER set_timestamp BEFORE UPDATE ON courses FOR EACH ROW EXECUTE FUNCTION trigger_set_timestamp()';
    EXECUTE 'CREATE TRIGGER set_timestamp BEFORE UPDATE ON gallery FOR EACH ROW EXECUTE FUNCTION trigger_set_timestamp()';
    EXECUTE 'CREATE TRIGGER set_timestamp BEFORE UPDATE ON media_assets FOR EACH ROW EXECUTE FUNCTION trigger_set_timestamp()';
    EXECUTE 'CREATE TRIGGER set_timestamp BEFORE UPDATE ON admins FOR EACH ROW EXECUTE FUNCTION trigger_set_timestamp()';
  END IF;
END$$;
